102    /// Computes a Balinese date from Gregorian year, month, day using an explicit day boundary.
103    ///
104    /// The Balinese calendar day begins at sunrise rather than midnight. This method allows
105    /// customizing how the sunrise is determined:
106    ///
107    /// - `DayBoundary::Midnight`: Uses Gregorian midnight (legacy behavior)
108    /// - `DayBoundary::FixedSunrise(hour)`: Uses a fixed hour offset (e.g., 6 for 06:00 UTC)
109    /// - `DayBoundary::Astronomical`: Calculates actual astronomical sunrise for given coordinates
110    ///
111    /// # Arguments
112    ///
113    /// * `year` - Gregorian year (1800-2200 CE)
114    /// * `month` - Gregorian month (1-12)
115    /// * `day` - Gregorian day (1-31, depending on month)
116    /// * `boundary` - The day boundary definition to apply
117    ///
118    /// # Example
119    ///
120    /// ```rust
121    /// use balinese_calendar::{BalineseDate, DayBoundary};
122    /// let boundary = DayBoundary::FixedSunrise(6);
123    /// let date = BalineseDate::from_ymd_with_boundary(2026, 3, 26, &boundary)?;
124    /// # Ok::<(), balinese_calendar::BalineseDateError>(())
125    /// ```
126    pub fn from_ymd_with_boundary(
127        year: i32,
128        month: u32,
129        day: u32,
130        boundary: &DayBoundary,
131    ) -> Result<Self, BalineseDateError> {
132        // Validate year range early for better error messages
133        if !(1800..=2200).contains(&year) {
134            return Err(BalineseDateError::OutOfRange);
135        }
136
137        // Convert Gregorian date to UTC datetime at midnight, then apply boundary
138        let naive_date = chrono::NaiveDate::from_ymd_opt(year, month, day)
139            .ok_or(BalineseDateError::InvalidDate { year, month, day })?;
140        let utc_datetime = naive_date
141            .and_hms_opt(0, 0, 0)
142            .ok_or(BalineseDateError::InvalidDate { year, month, day })?
143            .and_utc();
144
145        Self::from_utc_datetime_with_boundary(utc_datetime, boundary)
146    }

156    /// Returns today's Balinese date using an explicit [`DayBoundary`].
157    ///
158    /// # Example
159    /// ```rust
160    /// use balinese_calendar::{BalineseDate, DayBoundary};
161    /// let date = BalineseDate::today_with_boundary(&DayBoundary::FixedSunrise(6))?;
162    /// # Ok::<(), balinese_calendar::BalineseDateError>(())
163    /// ```
164    pub fn today_with_boundary(boundary: &DayBoundary) -> Result<Self, BalineseDateError> {
165        use chrono::Utc;
166        let utc_now = Utc::now();
167        Self::from_utc_datetime_with_boundary(utc_now, boundary)
168    }

170    /// Computes a Balinese date from a provided UTC datetime using an explicit [`DayBoundary`].
171    ///
172    /// This helper enables deterministic tests by injecting a fixed UTC instant.
173    pub(crate) fn from_utc_datetime_with_boundary(
174        utc_now: chrono::DateTime<chrono::Utc>,
175        boundary: &DayBoundary,
176    ) -> Result<Self, BalineseDateError> {
177        let date = match boundary {
178            DayBoundary::Midnight => {
179                (utc_now + chrono::Duration::hours(Self::BALI_UTC_OFFSET_HOURS)).date_naive()
180            }
181            DayBoundary::FixedSunrise(hour) => {
182                if *hour > 23 {
183                    return Err(BalineseDateError::InvalidBoundaryHour(*hour));
184                }
185                let offset_hours = Self::BALI_UTC_OFFSET_HOURS - (*hour as i64);
186                (utc_now + chrono::Duration::hours(offset_hours)).date_naive()
187            }
188            #[cfg(feature = "astronomical")]
189            DayBoundary::Astronomical { lat, lon } => {
190                // Calculate astronomical sunrise for the given coordinates
191                use sunrise::{Coordinates, SolarDay, SolarEvent};
192
193                let utc_date = utc_now.date_naive();
194                let coordinates = Coordinates::new(*lat, *lon)
195                    .ok_or(BalineseDateError::AstronomicalCalculationFailed)?;
196                let solar_day = SolarDay::new(coordinates, utc_date);
197
198                // Calculate sunrise time in UTC for the given date and coordinates
199                let sunrise_datetime = solar_day
200                    .event_time(SolarEvent::Sunrise)
201                    .ok_or(BalineseDateError::AstronomicalCalculationFailed)?;
202
203                // The Balinese day starts at astronomical sunrise.
204                // Determine which calendar day we're in based on whether utc_now
205                // is before or after today's sunrise.
206                if utc_now < sunrise_datetime {
207                    // Before today's sunrise, we're still in yesterday's Balinese day
208                    (sunrise_datetime - chrono::Duration::days(1)).date_naive()
209                } else {
210                    // After today's sunrise, we're in today's Balinese day
211                    sunrise_datetime.date_naive()
212                }
213            }
214        };
215        Self::from_naive_date(date)
216    }

220    fn from_jdn_unchecked(jdn: i64, year: i32, month: u32, day: u32) -> Self {
221        // Compute pawukon_day once and reuse for performance optimization
222        let pawukon_day = pawukon_day_raw(jdn);
223
224        // Wewaran - pass pawukon_day to avoid recomputation
225        let ekawara = Ekawara::from_jdn(jdn);
226        let dwiwara = Dwiwara::from_jdn(jdn);
227        let triwara = Triwara::from_jdn(jdn);
228        let caturwara = Caturwara::from_jdn(jdn);
229        let pancawara = Pancawara::from_jdn(jdn);
230        let sadwara = Sadwara::from_jdn(jdn);
231        let saptawara = Saptawara::from_jdn(jdn);
232        let astawara = Astawara::from_jdn(jdn);
233        let sangawara = Sangawara::from_jdn(jdn);
234        let dasawara = Dasawara::from_wewaran(&pancawara, &saptawara);
235
236        // Pawukon - reuse computed pawukon_day
237        let wuku = Wuku::from_jdn(jdn);
238        let wuku_day = wuku_day_of_week(jdn);
239
240        // Paringkelan - use optimized constructors with precomputed pawukon_day
241        let jejepan = Jejepan::from_pawukon_day(pawukon_day);
242        let ingkel = Ingkel::from_pawukon_day(pawukon_day);
243        let watek_madya = WatekMadya::from_pawukon_day(pawukon_day);
244        let watek_alit = WatekAlit::from_pawukon_day(pawukon_day);
245        let lintang = Lintang::from_wewaran(&pancawara, &saptawara);
246        let panca_suda = PancaSuda::from_wewaran(&pancawara, &saptawara);
247        let pararasan = Pararasan::from_wewaran(&pancawara, &saptawara);
248        let rakam = Rakam::from_wewaran(&pancawara, &saptawara);
249
250        // Sasih
251        let sasih_result = SasihResult::from_jdn(jdn);
252        let is_purnama = sasih_result.day_info.is_purnama();
253        let is_tilem = sasih_result.day_info.is_tilem();
254
255        // Rahinan
256        let rahinan = Rahinan::detect(
257            &wuku,
258            &pancawara,
259            &saptawara,
260            &triwara,
261            &sasih_result.sasih,
262            &sasih_result.day_info,
263        );
264
265        BalineseDate {
266            gregorian_year: year,
267            gregorian_month: month,
268            gregorian_day: day,
269            jdn,
270            pawukon_day, // Store the computed value
271            wuku,
272            wuku_day,
273            ekawara,
274            dwiwara,
275            triwara,
276            caturwara,
277            pancawara,
278            sadwara,
279            saptawara,
280            astawara,
281            sangawara,
282            dasawara,
283            jejepan,
284            ingkel,
285            watek_madya,
286            watek_alit,
287            lintang,
288            panca_suda,
289            pararasan,
290            rakam,
291            saka_year: sasih_result.saka_year,
292            sasih: sasih_result.sasih,
293            sasih_day: sasih_result.day_info,
294            is_nampih: sasih_result.is_nampih,
295            is_purnama,
296            is_tilem,
297            rahinan,
298        }
299    }

303    /// Full traditional Balinese date string.
304    /// Format: "Saptawara Pancawara Wuku, Tithi Sasih Saka-year"
305    /// Example: "Kamis Umanis Sungsang, Penanggal 15 Kasanga 1948"
306    pub fn to_balinese_string(&self) -> String {
307        let tithi = match self.sasih_day {
308            SasihDayInfo::Single(phase) => phase.to_string(),
309            SasihDayInfo::Ngunaratri { primary, .. } => format!("Ngunaratri ({primary})"),
310        };
311
312        format!(
313            "{} {} {}, {} {} Saka {}",
314            self.saptawara.name(),
315            self.pancawara.name(),
316            self.wuku.name(),
317            tithi,
318            self.sasih.name(),
319            self.saka_year,
320        )
321    }

374    /// 2026-03-08 00:30 WITA = 2026-03-07 16:30 UTC
375    /// With FixedSunrise(6): UTC+2 offset → date = 2026-03-07
376    /// With Midnight:        UTC+8 offset → date = 2026-03-08 (Gregorian)
377    #[test]
378    fn fixed_sunrise_before_dawn_returns_prior_gregorian_day() {
379        let utc = chrono::DateTime::parse_from_rfc3339("2026-03-07T16:30:00Z").unwrap();
380        let date = BalineseDate::from_utc_datetime_with_boundary(
381            utc.with_timezone(&chrono::Utc),
382            &DayBoundary::FixedSunrise(6),
383        )
384        .unwrap();
385        assert_eq!(date.gregorian_year, 2026);
386        assert_eq!(date.gregorian_month, 3);
387        assert_eq!(date.gregorian_day, 7);
388    }

460/// Flat record struct for columnar data serialization.
461/// Suitable for Arrow, Parquet, and other columnar formats.
462///
463/// # Serde Limitations
464///
465/// When the `serde` feature is enabled, this struct can be serialized to JSON
466/// but **cannot be deserialized** from JSON due to the `&'static str` fields
467/// (`sasih_name`, `sasih_season_tag`, etc.). This is a design constraint
468/// required for zero-copy string references. For API integration, use this
469/// struct for JSON output only.
470#[derive(Debug, Clone, PartialEq)]
471#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
472pub struct FlatRecord {
473    pub gregorian_year: i32,
474    pub gregorian_month: u32,
475    pub gregorian_day: u32,
476    pub jdn: i64,
477    pub saka_year: i32,
478    pub sasih_id: u8,
479    pub sasih_name: &'static str,
480    pub sasih_season_tag: &'static str,
481    pub pancaroba_flag: bool,
482    pub is_purnama: bool,
483    pub is_tilem: bool,
484    pub is_ngunaratri: bool,
485    pub is_nampih: bool,
486    pub tithi: u8,
487    pub pawukon_day: u16,
488    pub wuku_id: u8,
489    pub wuku_name: &'static str,
490    pub wuku_ecology_tag: &'static str,
491    pub pancawara_id: u8,
492    pub pancawara_name: &'static str,
