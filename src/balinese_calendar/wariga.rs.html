22/// Wariga BELOG — personalized day quality based on birth date and query date.
23///
24/// The simplest Dewasa Ayu feature: personalized to birth date, computable today.
25/// Uses pure modular arithmetic: `(birth_urip + daily_urip) % 4` where urip = sapta + panca.
26///
27/// Source: Wariga BELOG manuscript (Gianyar tradition), via T.I.P. Nyoman (2014)
28/// *Guide Book Buku Pedoman Wariga Belog*, Koleksi Griya Cebaang Giri Kesuma.
29#[derive(Debug, Clone, Copy, PartialEq, Eq)]
30#[cfg_attr(feature = "serde", derive(serde::Serialize, serde::Deserialize))]
31pub enum WarigaBelog {
32    /// Pati — danger, avoid major activities
33    Pati = 0,
34    /// Guru — wisdom, good for learning/spiritual practice
35    Guru = 1,
36    /// Ratu — authority, good for leadership/official matters
37    Ratu = 2,
38    /// Lara — suffering, avoid important undertakings
39    Lara = 3,
40}
41
42impl WarigaBelog {
43    /// Get descriptive text for each Wariga BELOG quality.
44    pub fn description(&self) -> &'static str {
45        match self {
46            WarigaBelog::Pati => "danger, avoid major activities",
47            WarigaBelog::Guru => "wisdom, good for learning/spiritual practice",
48            WarigaBelog::Ratu => "authority, good for leadership/official matters",
49            WarigaBelog::Lara => "suffering, avoid important undertakings",
50        }
51    }
52}

54/// Calculate Wariga BELOG personalized day quality.
55///
56/// Algorithm: `(birth_urip + daily_urip) % 4` where urip = sapta_wara.urip() + panca_wara.urip().
57///
58/// # Arguments
59/// * `birth` - Birth date for personalization
60/// * `query` - Query date to check quality for
61///
62/// # Returns
63/// * `WarigaBelog` - Personalized day quality (Pati, Guru, Ratu, or Lara)
64pub fn wariga_belog(birth: &BalineseDate, query: &BalineseDate) -> WarigaBelog {
65    let birth_urip = birth.saptawara.urip() + birth.pancawara.urip();
66    let query_urip = query.saptawara.urip() + query.pancawara.urip();
67    let combined = (birth_urip + query_urip) % 4;
68
69    match combined {
70        0 => WarigaBelog::Pati,
71        1 => WarigaBelog::Guru,
72        2 => WarigaBelog::Ratu,
73        3 => WarigaBelog::Lara,
74        _ => unreachable!(), // modulo 4 ensures 0-3 range
75    }
76}

140/// Load Tri-Pramana lookup table from JSON fixture.
141///
142/// Returns the complete 210-entry lookup table mapping each Wuku-day position
143/// to its Tri-Pramana urip value and quality classification.
144///
145/// The JSON data is loaded from the fixture file containing the traditional
146/// values extracted from I Made Bidja's Wariga Sundari Bungkah manuscript.
147fn load_tri_pramana_lookup() -> Result<[TriPramana; 210], Box<dyn std::error::Error>> {
148    // Try to load from the JSON fixture file
149    let json_path =
150        concat!(env!("CARGO_MANIFEST_DIR"), "/tests/fixtures/gebogan_urip_tri_pramana.json");
151
152    let json_content = fs::read_to_string(json_path)?;
153    let data: serde_json::Value = serde_json::from_str(&json_content)?;
154
155    let entries = data["entries"]
156        .as_array()
157        .ok_or("Invalid JSON structure: missing entries array")?;
158
159    if entries.len() != 210 {
160        return Err(format!("Expected 210 entries, found {}", entries.len()).into());
161    }
162
163    let mut lookup = [TriPramana { urip: 1, quality: PramanaQuality::LungguhSakti }; 210];
164
165    for (i, entry) in entries.iter().enumerate() {
166        let urip =
167            entry["tri_pramana_urip"].as_u64().ok_or("Invalid tri_pramana_urip value")? as u8;
168
169        // Validate urip range (should be 1-30)
170        if urip == 0 || urip > 30 {
171            return Err(format!("Invalid urip value {} at index {}", urip, i).into());
172        }
173
174        lookup[i] = TriPramana::from_urip(urip);
175    }
176
177    Ok(lookup)
178}

180/// Get Tri-Pramana data for a specific Pawukon day position.
181///
182/// # Arguments
183/// * `pawukon_day` - Absolute position in 210-day cycle (0-209)
184///
185/// # Returns
186/// * `Option<TriPramana>` - Composite urip value and quality classification,
187///   or None if pawukon_day is out of bounds
188///
189/// # Panics
190/// This function will panic if the JSON fixture cannot be loaded or parsed.
191/// This is intentional to ensure the traditional data is always available.
192pub fn tri_pramana_for_day(pawukon_day: u16) -> Option<TriPramana> {
193    if pawukon_day >= 210 {
194        return None;
195    }
196
197    static LOOKUP: std::sync::OnceLock<
198        Result<[TriPramana; 210], Box<dyn std::error::Error + Send + Sync>>,
199    > = std::sync::OnceLock::new();
200
201    let lookup_result = LOOKUP.get_or_init(|| {
202        // Try to load from JSON fixture first
203        match load_tri_pramana_lookup() {
204            Ok(data) => Ok(data),
205            Err(e) => {
206                // Fallback to simple algorithm if JSON loading fails
207                eprintln!(
208                    "Warning: Failed to load Tri-Pramana JSON data: {}. Using fallback algorithm.",
209                    e
210                );
211                let mut fallback =
212                    [TriPramana { urip: 1, quality: PramanaQuality::LungguhSakti }; 210];
213                for (i, item) in fallback.iter_mut().enumerate() {
214                    let urip = ((i % 30) + 1) as u8;
215                    *item = TriPramana::from_urip(urip);
216                }
217                Ok(fallback)
218            }
219        }
220    });
221
222    match lookup_result.as_ref() {
223        Ok(lookup) => Some(lookup[pawukon_day as usize]),
224        Err(_) => None,
225    }
226}

527/// Calculate Otonan (Balinese birthday) dates.
528///
529/// The otonan falls every 210 days (one complete Pawukon cycle).
530///
531/// # Arguments
532/// * `birth` - Birth date
533/// * `count` - Number of future otonan dates to calculate
534///
535/// # Returns
536/// * `Vec<NaiveDate>` - Future otonan dates
537pub fn otonan_dates(birth: NaiveDate, count: usize) -> Vec<NaiveDate> {
538    let mut dates = Vec::with_capacity(count);
539    let cycle_duration = Duration::days(210);
540
541    for i in 1..=count {
542        dates.push(birth + cycle_duration * i as i32);
543    }
544
545    dates
546}

548/// Get the next Otonan date after today.
549///
550/// This is a convenience function that uses the current system date.
551/// For deterministic behavior in tests, use `next_otonan_from` with an explicit date.
552///
553/// # Arguments
554/// * `birth` - Birth date
555///
556/// # Returns
557/// * `NaiveDate` - Next otonan date after today
558pub fn next_otonan(birth: NaiveDate) -> NaiveDate {
559    let today = chrono::Utc::now().date_naive();
560    next_otonan_from(birth, today)
561}
562
563/// Get the next Otonan date after a specific reference date.
564///
565/// This function is deterministic and suitable for testing.
566///
567/// # Arguments
568/// * `birth` - Birth date
569/// * `after` - Reference date (calculate otonan after this date)
570///
571/// # Returns
572/// * `NaiveDate` - Next otonan date after the reference date
573pub fn next_otonan_from(birth: NaiveDate, after: NaiveDate) -> NaiveDate {
574    let cycle_duration = Duration::days(210);
575
576    // Find the first otonan after the reference date
577    let mut candidate = birth;
578
579    while candidate <= after {
580        candidate += cycle_duration;
581    }
582
583    candidate
584}
