1// src/sasih.rs
2//
3// Sasih — Balinese lunar month system.
4// This is the most algorithmically complex part of the Balinese calendar.
5//
6// Key concepts:
7//   Penanggal  — waxing phase (tithi 1–15), starts day after Tilem (new moon)
8//   Pangelong  — waning phase (tithi 1–15), starts day after Purnama (full moon)
9//   Purnama    — full moon (end of penanggal 15)
10//   Tilem      — new moon (end of pangelong 15)
11//   Ngunaratri — "minus one night": every 63 solar days, a lunar day is skipped
12//                (two lunar days fall on one solar day)
13//   Nampih Sasih — intercalary (13th) month inserted to re-sync with solar year
14//
15// Algorithm basis:
16//   - peradnya/balinese-date-js-lib (Apache-2.0) — walk-forward pivot algorithm
17//   - Pendit, N.S. (2001). "Nyepi: kebangkitan, toleransi, dan kerukunan."
18//   - babadbali.com Sasih algorithm
19//
20// The algorithm uses two pivot points (known dates with all sasih parameters)
21// and walks forward or backward to the target date, counting sasih boundaries
22// and determining nampih (intercalary) months algorithmically from the
23// 19-year Metonic-like cycle (saka_year % 19).
24//
25// ⚠ PRODUCTION NOTE:
26//   The Nampih Sasih placement computed algorithmically may differ from the
27//   official PHDI calendar by 1 sasih in some years. Verify critical dates
28//   against the PHDI calendar or kalenderbali.org.
29
30use crate::utils::{NGUNARATRI_PERIOD, SAKA_YEAR_OFFSET};

34#[derive(Debug, Clone, Copy, PartialEq, Eq)]
35pub enum Sasih {
36    Kasa = 0,
37    Karo = 1,
38    Katiga = 2,
39    Kapat = 3,
40    Kalima = 4,
41    Kanem = 5,
42    Kapitu = 6,
43    Kawolu = 7,
44    Kasanga = 8,
45    Kadasa = 9,
46    Desta = 10, // Jyestha
47    Sada = 11,  // Sadha
48    // Intercalary months (Nampih Sasih):
49    NampihDesta = 12, // Mala/Nampih Jyestha — inserted after Desta
50    NampihSada = 13,  // Mala/Nampih Sadha   — inserted after Sada
51}
52
53impl Sasih {
54    pub fn name(&self) -> &'static str {
55        match self {
56            Sasih::Kasa => "Kasa",
57            Sasih::Karo => "Karo",
58            Sasih::Katiga => "Katiga",
59            Sasih::Kapat => "Kapat",
60            Sasih::Kalima => "Kalima",
61            Sasih::Kanem => "Kanem",
62            Sasih::Kapitu => "Kapitu",
63            Sasih::Kawolu => "Kawolu",
64            Sasih::Kasanga => "Kasanga",
65            Sasih::Kadasa => "Kadasa",
66            Sasih::Desta => "Desta",
67            Sasih::Sada => "Sada",
68            Sasih::NampihDesta => "Nampih Desta",
69            Sasih::NampihSada => "Nampih Sada",
70        }
71    }
72
73    /// Traditional season association
74    ///
75    /// Based on academic validation from lontar texts, peer-reviewed research, and ritual tradition:
76    /// - Purwaputera et al. (2025), IJMRA Vol. 08 Issue 08 (Udayana/BMKG)
77    /// - Ginaya (2018), IJLLC 4(3), 24-37
78    /// - Lontar Purwaka Bumi, Cuda Mani, Purana Bali Dwipa
79    /// - Nangluk Merana / Caru Sasih Kalima–Kanem ritual tradition
80    ///
81    /// Pancaroba (transitional season, dry→wet) spans Sasih Kalima and Sasih Kanem,
82    /// approximately October–December Gregorian. This is the ONLY academically validated
83    /// pancaroba in the Balinese Saka Calendar.
84    ///
85    /// Season mapping:
86    /// - Dry (kemarau): Jyesta, Sadha, Kasa, Karo, Katiga, Kapat (≈Apr-Oct)
87    /// - Pancaroba (transitional): Kalima, Kanem (≈Oct-Dec)
88    /// - Wet (hujan): Kapitu, Kawulu, Kasanga, Kadasa (≈Dec-Mar)
89    pub fn season_tag(&self) -> &'static str {
90        match self {
91            Sasih::Kasa => "dry",         // Peak dry season (June–Aug)
92            Sasih::Karo => "dry",         // Peak dry season
93            Sasih::Katiga => "dry",       // Dry season
94            Sasih::Kapat => "dry",        // Late dry, "blabur" possible
95            Sasih::Kalima => "pancaroba", // Early transition, onset of rain
96            Sasih::Kanem => "pancaroba",  // Full transition, Nangluk Merana period
97            Sasih::Kapitu => "wet",       // Peak wet season (Dec–Feb)
98            Sasih::Kawolu => "wet",       // Peak wet season
99            Sasih::Kasanga => "wet",      // Late wet season
100            Sasih::Kadasa => "wet",       // Waning wet, pre-Nyepi
101            Sasih::Desta => "dry",        // Early dry season (Jyesta)
102            Sasih::Sada => "dry",         // Dry season
103            Sasih::NampihDesta => "dry",  // Intercalary month follows dry season
104            Sasih::NampihSada => "dry",   // Intercalary month follows dry season
105        }
106    }
107
108    /// Returns true if this sasih falls in the traditional pancaroba period.
109    ///
110    /// The pancaroba (transitional season, dry→wet) spans Sasih Kalima and
111    /// Sasih Kanem, approximately October–December Gregorian. This is the
112    /// only academically and ritually validated pancaroba in the Balinese
113    /// Saka Calendar.
114    ///
115    /// Sources:
116    /// - Lontar Purwaka Bumi, Lontar Cuda Mani, Purana Bali Dwipa
117    /// - Nangluk Merana / Caru Sasih Kalima–Kanem ceremony tradition
118    /// - Purwaputera et al. (2025), IJMRA Vol. 08 Issue 08
119    /// - Ginaya (2018), IJLLC 4(3), 24-37
120    /// - I Wayan Tusan (1974), Ala Ayuning Sasih / Wariga tradition
121    ///
122    /// Note: There is NO traditional second pancaroba for wet→dry transition.
123    /// BMKG meteorological classification (April pancaroba) ≠ Balinese tradition.
124    pub fn is_pancaroba(&self) -> bool {
125        matches!(self, Sasih::Kalima | Sasih::Kanem)
126    }
127
128    pub fn is_planting_signal(&self) -> bool {
129        matches!(self, Sasih::Katiga | Sasih::Kapat | Sasih::Kalima)
130    }
131
132    fn from_index(idx: i32) -> Self {
133        match ((idx % 12) + 12) % 12 {
134            0 => Sasih::Kasa,
135            1 => Sasih::Karo,
136            2 => Sasih::Katiga,
137            3 => Sasih::Kapat,
138            4 => Sasih::Kalima,
139            5 => Sasih::Kanem,
140            6 => Sasih::Kapitu,
141            7 => Sasih::Kawolu,
142            8 => Sasih::Kasanga,
143            9 => Sasih::Kadasa,
144            10 => Sasih::Desta,
145            _ => Sasih::Sada,
146        }
147    }
148}

152#[derive(Debug, Clone, Copy, PartialEq, Eq)]
153pub enum TithiPhase {
154    /// Waxing moon — day 1–14 (Penanggal 1–14)
155    Penanggal(u8),
156    /// Full moon (Penanggal 15 = Purnama)
157    Purnama,
158    /// Waning moon — day 1–14 (Pangelong 1–14)
159    Pangelong(u8),
160    /// New moon (Pangelong 15 = Tilem)
161    Tilem,
162}
163
164impl TithiPhase {
165    pub fn tithi_number(&self) -> u8 {
166        match self {
167            TithiPhase::Penanggal(n) => *n,
168            TithiPhase::Purnama => 15,
169            TithiPhase::Pangelong(n) => 15 + n,
170            TithiPhase::Tilem => 30,
171        }
172    }
173}
174
175impl std::fmt::Display for TithiPhase {
176    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
177        match self {
178            TithiPhase::Penanggal(n) => write!(f, "Penanggal {n}"),
179            TithiPhase::Purnama => write!(f, "Purnama"),
180            TithiPhase::Pangelong(n) => write!(f, "Pangelong {n}"),
181            TithiPhase::Tilem => write!(f, "Tilem"),
182        }
183    }
184}

186// ── SasihDayInfo — phase within the lunar month, including ngunaratri ────────

188#[derive(Debug, Clone, Copy, PartialEq, Eq)]
189pub enum SasihDayInfo {
190    /// Single tithi on this solar day
191    Single(TithiPhase),
192    /// Ngunaratri — two lunar days on one solar day; primary + secondary tithi.
193    Ngunaratri {
194        primary: TithiPhase,
195        secondary: TithiPhase,
196    },
197}
198
199impl std::fmt::Display for SasihDayInfo {
200    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
201        match self {
202            SasihDayInfo::Single(TithiPhase::Penanggal(n)) => write!(f, "Penanggal {n}"),
203            SasihDayInfo::Single(TithiPhase::Purnama) => write!(f, "Purnama"),
204            SasihDayInfo::Single(TithiPhase::Pangelong(n)) => write!(f, "Pangelong {n}"),
205            SasihDayInfo::Single(TithiPhase::Tilem) => write!(f, "Tilem"),
206            SasihDayInfo::Ngunaratri { primary, .. } => write!(f, "Ngunaratri ({primary})"),
207        }
208    }
209}
210
211impl SasihDayInfo {
212    pub fn is_purnama(&self) -> bool {
213        matches!(
214            self,
215            SasihDayInfo::Single(TithiPhase::Purnama)
216                | SasihDayInfo::Ngunaratri {
217                    primary: TithiPhase::Purnama,
218                    ..
219                }
220        )
221    }
222    pub fn is_tilem(&self) -> bool {
223        matches!(
224            self,
225            SasihDayInfo::Single(TithiPhase::Tilem)
226                | SasihDayInfo::Ngunaratri {
227                    primary: TithiPhase::Tilem,
228                    ..
229                }
230        )
231    }
232    pub fn is_ngunaratri(&self) -> bool {
233        matches!(self, SasihDayInfo::Ngunaratri { .. })
234    }
235
236    pub fn tithi_number(&self) -> u8 {
237        match self {
238            SasihDayInfo::Single(phase) => phase.tithi_number(),
239            SasihDayInfo::Ngunaratri { primary, .. } => primary.tithi_number(),
240        }
241    }
242}

254// ── Pivot points ──────────────────────────────────────────────────────────────
255// Known dates with fully characterized sasih parameters.
256// Source: peradnya/balinese-date-js-lib (Apache-2.0).
257
258struct Pivot {
259    jdn: i64,
260    sasih_day: i64,      // day within the 30-day sasih at this pivot (0-based)
261    ngunaratri_day: i64, // ngunaratri offset at this pivot
262    saka: i32,
263    sasih_id: i32, // 0-based sasih index (Kasa=0 .. Sada=11)
264    is_nampih: bool,
265}
266
267// PIVOT_1971: January 27, 1971 — Penanggal 1 Kapitu Saka 1892
268const PIVOT_1971: Pivot = Pivot {
269    jdn: 2_440_979, // gregorian_to_jdn(1971, 1, 27)
270    sasih_day: 0,
271    ngunaratri_day: 0,
272    saka: 1892,
273    sasih_id: 6, // Kapitu
274    is_nampih: false,
275};
276
277// PIVOT_2000: January 18, 2000 — Pangelong 12 Kapitu Saka 1921
278const PIVOT_2000: Pivot = Pivot {
279    jdn: 2_451_562, // gregorian_to_jdn(2000, 1, 18)
280    sasih_day: 12,
281    ngunaratri_day: 0,
282    saka: 1921,
283    sasih_id: 6, // Kapitu
284    is_nampih: false,
285};

287// Sasih Kesinambungan (SK) period boundaries (JDN)
288// During this period (1993-01-24 to 2003-01-03), different nampih rules apply.
289const SK_START_JDN: i64 = 2_448_998; // 1993-01-24
290const SK_END_JDN: i64 = 2_452_642; // 2003-01-03
291
292// ── Euclidean modulo helper ──────────────────────────────────────────────────
293fn emod(a: i64, b: i64) -> i64 {
294    ((a % b) + b) % b
295}
296
297// ── Core calculation ──────────────────────────────────────────────────────────
298
299impl SasihResult {
300    pub fn from_jdn(jdn: i64) -> Self {
301        // Select the best pivot point
302        let pivot = if jdn < 2_451_550 {
303            // Before ~Jan 6, 2000 (Pangalantaka Paing)
304            &PIVOT_1971
305        } else {
306            &PIVOT_2000
307        };
308
309        let day_diff = jdn - pivot.jdn;
310
311        // ── 1. Compute sasih day info (tithi, phase, ngunaratri) ─────────
312        let day_skip = if day_diff >= 0 {
313            // ceil(day_diff / 63) for positive
314            (day_diff + NGUNARATRI_PERIOD - 1) / NGUNARATRI_PERIOD
315        } else {
316            // ceil for negative = -floor(abs/63)
317            -((-day_diff) / NGUNARATRI_PERIOD)
318        };
319        let day_total = pivot.sasih_day + day_diff + day_skip;
320
321        let raw_day = emod(day_total, 30) as u8; // 0–29
322        let is_pangelong = raw_day == 0 || raw_day > 15;
323        let is_ngunaratri = emod(day_diff, NGUNARATRI_PERIOD) == 0 && day_diff != 0;
324
325        let tithi_in_phase = emod(day_total, 15) as u8; // 0–14
326        let tithi = if tithi_in_phase == 0 {
327            15
328        } else {
329            tithi_in_phase
330        };
331
332        let day_info = Self::build_day_info(tithi, is_pangelong, is_ngunaratri);
333
334        // ── 2. Walk-forward to determine sasih and saka year ─────────────
335        let pivot_offset = if pivot.sasih_day == 0 && pivot.ngunaratri_day == 0 {
336            0
337        } else {
338            1
339        };
340
341        let total_sasih = if day_diff >= 0 {
342            (day_total + 29) / 30 - pivot_offset // ceil(day_total / 30) - offset
343        } else {
344            -((-day_total) / 30) - pivot_offset
345        };
346
347        let mut current_sasih = pivot.sasih_id;
348        let mut current_saka = pivot.saka - if current_sasih == 9 { 1 } else { 0 }; // Kadasa offset
349        let mut nampih_count: i32 = if pivot.is_nampih { 1 } else { 0 };
350        let mut in_sk = pivot.jdn >= SK_START_JDN && pivot.jdn < SK_END_JDN;
351
352        let mut remaining = total_sasih;
353
354        while remaining != 0 {
355            if day_diff >= 0 {
356                // Walking forward
357                if nampih_count == 0 || nampih_count == 2 {
358                    nampih_count = 0;
359                    current_sasih = emod(current_sasih as i64 + 1, 12) as i32;
360                }
361                remaining -= 1;
362
363                if current_sasih == 9 && nampih_count == 0 {
364                    // Kadasa = start of new Saka year
365                    current_saka += 1;
366                }
367
368                // SK period tracking
369                if current_sasih == 7 && current_saka == 1914 {
370                    in_sk = true;
371                } else if current_sasih == 7 && current_saka == 1924 {
372                    in_sk = false;
373                }
374            } else {
375                // Walking backward
376                if nampih_count == 0 || nampih_count == 2 {
377                    nampih_count = 0;
378                    current_sasih = emod(current_sasih as i64 - 1, 12) as i32;
379                }
380                remaining += 1;
381
382                if current_sasih == 8 && nampih_count == 0 {
383                    // Kasanga going backward = leaving this Saka year
384                    current_saka -= 1;
385                }
386
387                // SK period tracking (reverse)
388                if current_sasih == 6 && current_saka == 1914 {
389                    in_sk = false;
390                } else if current_sasih == 6 && current_saka == 1924 {
391                    in_sk = true;
392                }
393            }
394
395            // Nampih detection via 19-year Metonic-like cycle
396            nampih_count += Self::nampih_increment(current_saka, current_sasih, in_sk);
397        }
398
399        let is_nampih = nampih_count == 1;
400        let sasih = if is_nampih {
401            match Sasih::from_index(current_sasih) {
402                Sasih::Desta => Sasih::NampihDesta,
403                Sasih::Sada => Sasih::NampihSada,
404                other => other, // fallback — shouldn't happen with correct algorithm
405            }
406        } else {
407            Sasih::from_index(current_sasih)
408        };
409
410        SasihResult {
411            saka_year: current_saka,
412            sasih,
413            day_info,
414            is_nampih,
415        }
416    }
417
418    /// Determine if the current sasih in the given saka year triggers a nampih.
419    /// Returns 1 if nampih should be inserted, 0 otherwise.
420    /// Based on the 19-year Metonic-like cycle from peradnya.
421    fn nampih_increment(saka: i32, sasih_id: i32, in_sk: bool) -> i32 {
422        let cycle = ((saka % 19) + 19) % 19;
423        match cycle {
424            0 | 6 | 11 => {
425                if sasih_id == 10 && !in_sk && saka != 1925 {
426                    1 // Desta
427                } else {
428                    0
429                }
430            }
431            3 | 8 | 14 | 16 => {
432                if sasih_id == 11 && !in_sk {
433                    1 // Sada
434                } else {
435                    0
436                }
437            }
438            // SK-period specific rules
439            2 | 10 => {
440                if sasih_id == 10 && in_sk {
441                    1 // Desta during SK
442                } else {
443                    0
444                }
445            }
446            4 => {
447                if sasih_id == 2 && in_sk {
448                    1 // Katiga during SK
449                } else {
450                    0
451                }
452            }
453            7 => {
454                if sasih_id == 0 && in_sk {
455                    1 // Kasa during SK
456                } else {
457                    0
458                }
459            }
460            13 => {
461                if sasih_id == 9 && in_sk {
462                    1 // Kadasa during SK
463                } else {
464                    0
465                }
466            }
467            15 => {
468                if sasih_id == 1 && in_sk {
469                    1 // Karo during SK
470                } else {
471                    0
472                }
473            }
474            _ => 0,
475        }
476    }
477
478    fn build_day_info(tithi: u8, is_pangelong: bool, is_ngunaratri: bool) -> SasihDayInfo {
479        let primary_phase = if is_pangelong {
480            if tithi == 15 || (tithi == 14 && is_ngunaratri) {
481                TithiPhase::Tilem
482            } else {
483                TithiPhase::Pangelong(tithi)
484            }
485        } else if tithi == 15 || (tithi == 14 && is_ngunaratri) {
486            TithiPhase::Purnama
487        } else {
488            TithiPhase::Penanggal(tithi)
489        };
490
491        if is_ngunaratri {
492            let next_tithi = if tithi == 15 { 1 } else { tithi + 1 };
493            let secondary_phase = if is_pangelong {
494                if next_tithi > 14 {
495                    TithiPhase::Tilem
496                } else {
497                    TithiPhase::Pangelong(next_tithi)
498                }
499            } else if next_tithi > 14 {
500                TithiPhase::Purnama
501            } else {
502                TithiPhase::Penanggal(next_tithi)
503            };
504            SasihDayInfo::Ngunaratri {
505                primary: primary_phase,
506                secondary: secondary_phase,
507            }
508        } else {
509            SasihDayInfo::Single(primary_phase)
510        }
511    }
512}
